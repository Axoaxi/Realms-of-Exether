##### Effects
consortium_approve_membership = {
	effect = {
		scope:applicant_country_scope = {
			add_to_variable_list = {
				name = consortium_supporting
				target = ROOT
			}
			change_variable = {
				name = consortium_supporting_list_size
				add = 1
			}
			if = {
				limit = {
					is_target_in_variable_list = {
						name = consortium_rejecting
						target = ROOT
					}
				}
				change_variable = {
					name = consortium_supporting_list_size
					add = 1
				}
			}
			remove_list_variable = {
				name = consortium_rejecting
				target = ROOT
			}
			remove_list_variable = {
				name = consortium_abstaining
				target = ROOT
			}
		}
	}
}
consortium_abstain_membership = {
	effect = {
		scope:applicant_country_scope = {
			add_to_variable_list = {
				name = consortium_abstaining
				target = ROOT
			}

					is_target_in_variable_list = {
						name = consortium_supporting
						target = ROOT
					}
				}

					name = consortium_supporting_list_size
					subtract = 1
				}
			}
			if = {
				limit = {
					is_target_in_variable_list = {
						name = consortium_rejecting
						target = ROOT
					}
				}
				change_variable = {
					name = consortium_supporting_list_size
					add = 1
				}
			}
			remove_list_variable = {
				name = consortium_supporting
				target = ROOT
			}
			remove_list_variable = {
				name = consortium_rejecting
				target = ROOT
			}
		}
	}
}

############# Declaration of Human Rights
# Varcheck
declaration_of_human_rights_varcheck = {
	is_shown = {
		NOR = {
			has_global_variable = pending_resolution
			has_global_variable = declaration_of_human_rightscomplete
			has_global_variable = declaration_of_human_rightsfail
		}
	}
}
# Execute
activate_declaration_of_human_rights_je = {
	effect = {
		every_country = {
			if = {
				limit = {
					or = {
						has_variable = consortium_major_member
						has_variable = consortium_member
					}
				}
				add_journal_entry = {
					type = je_declaration_of_human_rights
				}
			}
		}
		set_global_variable = pending_resolution
		set_global_variable = {
			name = pending_resolution_progress
			value = 0
		}
	}
}
