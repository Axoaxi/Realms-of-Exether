# This is executed last among all history

	}

	# Jack the ripper, prevents the event from happening every game

	every_country = {
		if = {
			limit = {

			}
		}
		random_list = {
			1 = { set_variable = var_jack_ripper_possible }

				#Nothing Happens
			}
		}
	}
